    SELECT 
        YEAR(OrderDate) AS SalesYear,
        MONTH(OrderDate) AS SalesMonth,
        SUM(OrderTotal) AS TotalSales,

QUERY 8: CTE WITH RANKING - Top Customers

WITH CustomerSpending AS (
    SELECT 
        CustomerID,
        CustomerName,
        SUM(OrderTotal) AS LifetimeValue,
        COUNT(OrderID) AS OrderCount
    FROM Orders
    GROUP BY CustomerID, CustomerName
),
RankedCustomers AS (
    SELECT 
        CustomerID,
        CustomerName,
        LifetimeValue,
        OrderCount,
        ROW_NUMBER() OVER (ORDER BY LifetimeValue DESC) AS SpendingRank
    FROM CustomerSpending
)
SELECT * FROM RankedCustomers WHERE SpendingRank <= 5;
